def address_lookup(card_address):
    """只匹配重庆地区码，支持xxxx格式"""
    x_count = card_address.lower().count('x')
    normalized_address = card_address.lower().replace('x', r'\d')
    pattern = re.compile(f'^{normalized_address}$')
    
    matched_areas = {}
    for code, name in chongqing_districts.items():
        if pattern.match(code):
            matched_areas[code] = name
    
    print(f"重庆地区码需补齐 {x_count} 位，共匹配 {len(matched_areas)} 个辖区")
    return matched_areas.keys()

def check_id_data(n):
    """验证出生日期有效性"""
    if len(n) < 14:
        return False
    try:
        year = int(n[6:10])
        month = int(n[10:12])
        day = int(n[12:14])
    except ValueError:
        return False
    
    # 合理年份范围（1900-当前年份）
    current_year = time.localtime().tm_year
    if year < 1900 or year > current_year:
        return False
    if month < 1 or month > 12:
        return False
    if day < 1 or day > 31:
        return False
    
    # 月份天数校验
    if month in {4, 6, 9, 11} and day > 30:
        return False
    if month == 2:
        is_leap = (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0)
        if day > 29 or (not is_leap and day > 28):
            return False
    return True

def check_id(n):
    """验证身份证校验码"""
    if len(n) != 18:
        return False
    weight = [7, 9, 10, 5, 8, 4, 2, 1, 6, 3, 7, 9, 10, 5, 8, 4, 2]
    check_code = ['1', '0', 'X', '9', '8', '7', '6', '5', '4', '3', '2']
    
    try:
        sum_val = sum(int(n[i]) * weight[i] for i in range(17))
    except ValueError:
        return False
    
    return check_code[sum_val % 11] == n[17]

            return

    print(f"身份证结构分析{gender_hint}")
    
    # 构建未知位的可能数字列表
    digits = []

            if i == 16:  # 性别位
